```bash
# Проверить доступность портов с внешнего сервера
nc -zv ваш_внешний_ip 443
nc -zv ваш_внешний_ip 80

# Проверить через curl с детализацией
curl -v http://ваш_внешний_ip
curl -v https://ваш_внешний_ip

# Проверить через telnet
telnet ваш_внешний_ip 80

# Затем введите:
HEAD / HTTP/1.0
Host: ваш_внешний_ip
```
